import RegistrationSchema from 'ember-osf-web/models/registration-schema';

export interface MirageRegistrationSchema1 {
    active: RegistrationSchema['active'];
    name: () => RegistrationSchema['name']; // eslint-disable-line no-restricted-globals
    schemaVersion: RegistrationSchema['schemaVersion'];
    schemaNoConflict: () => RegistrationSchema['schema'];
}

export type MirageFactory<T> = {
    [P in keyof T]: T[P] | (() => T[P]);
};

export type MirageRegistrationSchema =
    Pick<MirageFactory<RegistrationSchema>, 'active' | 'name' | 'schemaVersion'> &
    {
        schemaNoConflict: () => RegistrationSchema['schema'];
    };
